    $f_{Z_1, Z_2}(z_1, z_2)=\frac{1}{2\pi\sigma_1\sigma_2\sqrt{1-\rho^2}}\exp(-\frac{\frac{(z_1-\mu_1)^2}{\sigma_1^2}-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)})$
    
    $f_{Z_1(z_1)}=\int_{-\infty}^{\infty}f_{Z_1, Z_2}(z_1, z_2)dz_2=\frac{1}{\sqrt{2\pi}\sigma_1}\exp(-\frac{(z_1-\mu_1)^2}{2\sigma_1^2})$
    
    $f_{Z_2|Z_1}(z_2|z_1)=\frac{f_{Z_1, Z_2}(z_1, z_2)}{f_{Z_1(z_1)}}$
    
    $=\frac{\frac{1}{2\pi\sigma_1\sigma_2\sqrt{1-\rho^2}}\exp(-\frac{\frac{(z_1-\mu_1)^2}{\sigma_1^2}-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)})}{\frac{1}{\sqrt{2\pi}\sigma_1}\exp(-\frac{(z_1-\mu_1)^2}{2\sigma_1^2})}$
    
    
    $=\frac{1}{2\pi\sigma_1\sigma_2\sqrt{1-\rho^2}} \cdot \sqrt{2\pi}\sigma_1 \cdot \exp(-\frac{\frac{(z_1-\mu_1)^2}{\sigma_1^2}-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)}) \exp(\frac{(z_1-\mu_1)^2}{2\sigma_1^2})$
    
    $=\frac{1}{\sqrt{2\pi}\sigma_2\sqrt{1-\rho^2}} \cdot \exp(-\frac{\frac{(z_1-\mu_1)^2}{\sigma_1^2}-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} + \frac{(z_1-\mu_1)^2}{2\sigma_1^2}) $
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} - \frac{\frac{(z_1-\mu_1)^2}{\sigma_1^2}}{2(1-\rho^2)} + \frac{(z_1-\mu_1)^2}{2\sigma_1^2})$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} - \frac{(z_1-\mu_1)^2}{2\sigma_1^2(1-\rho^2)} + \frac{(z_1-\mu_1)^2(1-\rho^2)}{2\sigma_1^2(1-\rho^2)}) $
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} + \frac{((1-\rho^2)-1)(z_1-\mu_1)^2}{2\sigma_1^2(1-\rho^2)}) $

    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} - \frac{\rho^2(z_1-\mu_1)^2}{2\sigma_1^2(1-\rho^2)} )$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{\frac{\rho^2(z_1-\mu_1)^2}{\sigma_1^2}-2\rho\frac{(z_1-\mu_1)(z_2-\mu_2)}{\sigma_1\sigma_2}+\frac{(z_2-\mu_2)^2}{\sigma_2^2}}{2(1-\rho^2)} )$
    
    Define $X=\frac{(z_1-\mu_1)}{\sigma_1}$ and $Y=\frac{(z_2-\mu_2)}{\sigma_2}$
    
    Then,
    
    $f_{Z_2|Z_1}(z_2|z_1)=\frac{f_{Z_1, Z_2}(z_1, z_2)}{f_{Z_1(z_1)}}$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{(X\rho)^2-2XY\rho+Y^2}{2(1-\rho^2)})$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{(X\rho-Y)^2}{2(1-\rho^2)})$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{(\frac{(z_1-\mu_1)}{\sigma_1}\rho-\frac{(z_2-\mu_2)}{\sigma_2})^2}{2(1-\rho^2)})$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{(\frac{(\rho\sigma_2(z_1-\mu_1)}{\sigma_1}-(z_2-\mu_2))^2}{2\sigma_2^2(1-\rho^2)})$
    
    $=\frac{1}{\sqrt{2\pi(\sigma_2^2(1-\rho^2))}} \cdot \exp(-\frac{(z_2-(\mu_2+\frac{\rho\sigma_2(z_1-\mu_1)}{\sigma_1}))^2}{2\sigma_2^2(1-\rho^2)})$
